package com.mufiid.pilwali2020.adapters

import android.annotation.SuppressLint
import android.content.Context
import android.text.Editable
import android.text.TextWatcher
import android.view.LayoutInflater
import android.view.View
import android.view.ViewGroup
import android.widget.EditText
import android.widget.ImageView
import android.widget.TextView
import androidx.recyclerview.widget.RecyclerView
import com.bumptech.glide.Glide
import com.mufiid.pilwali2020.R
import com.mufiid.pilwali2020.adapters.PaslonAdptr.MyViewHolder
import com.mufiid.pilwali2020.models.Paslon
import java.util.*

class PaslonAdapter_back(
    editModelArrayList: ArrayList<Paslon>?
) : RecyclerView.Adapter<MyViewHolder>() {
    private val inflater: LayoutInflater

    override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): MyViewHolder {
        val view = inflater.inflate(R.layout.item_paslon, parent, false)
        return MyViewHolder(view)
    }

    @SuppressLint("SetTextI18n")
    override fun onBindViewHolder(holder: PaslonAdapter_back.MyViewHolder, position: Int) {
        holder.editText.setText(editModelArrayList!![position].jumlah_suara)
        holder.namaPaslon.text = editModelArrayList!![position].noPeserta + ". " + editModelArrayList!![position].nmPeserta
        Glide.with(holder.itemView)
            .load(editModelArrayList!![position].foto)
            .into(holder.gambar)
        if (position == 1) {
            holder.view.visibility = View.GONE
        }
    }

    override fun getItemCount(): Int {
        return editModelArrayList!!.size
    }

    inner class MyViewHolder(itemView: View) :
        RecyclerView.ViewHolder(itemView) {
        var editText: EditText
        var namaPaslon: TextView
        var gambar: ImageView
        var view: View

        init {
            view = itemView.findViewById(R.id.divider) as View
            editText = itemView.findViewById<View>(R.id.et_suara_paslon) as EditText
            namaPaslon = itemView.findViewById<View>(R.id.nama_paslon) as TextView
            gambar =
                itemView.findViewById<View>(R.id.img_paslon) as ImageView
            editText.addTextChangedListener(object : TextWatcher {
                override fun beforeTextChanged(
                    charSequence: CharSequence,
                    i: Int,
                    i1: Int,
                    i2: Int
                ) {
                }

                override fun onTextChanged(
                    charSequence: CharSequence,
                    i: Int,
                    i1: Int,
                    i2: Int
                ) {
                    editModelArrayList!![adapterPosition].jumlah_suara =
                        editText.text.toString()
                }

                override fun afterTextChanged(editable: Editable) {}
            })
        }
    }

    companion object {
        var editModelArrayList: List<Paslon>? = null
    }

    init {
        inflater = LayoutInflater.from(ctx)
        Companion.editModelArrayList = editModelArrayList
    }
}
